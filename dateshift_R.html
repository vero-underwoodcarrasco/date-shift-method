Date Shift

#new column for date shift integer
newchildtbl$dateshift <- sample(-21:21              # the range
                                , nrow(newchildtbl) # number of rows to generate
                                , replace=T)        # with replacement (allow numbers to be reused)

newchildtbl

1     0000      2016-01-25      2016-09-23        10
6     2906      2016-06-17      2016-09-24        18
3     4142      2017-04-30      2018-01-30        14
4     4290      2016-01-17      2016-03-02        19
5     6563      2018-08-03      2018-09-17        -7
2     6834      2017-03-24      2018-08-06        -2

Apply Date Shift to Original Dates

newchildtbl$New_Date_1 <- newchildtbl$Original_Date_1 + newchildtbl$dateshift
newchildtbl$New_Date_2 <- newchildtbl$Original_Date_2 + newchildtbl$dateshift

Reorder and select needed columns

newchildtbl <- newchildtbl %>%
                    select(Study_ID, Original_Date_1, Original_Date_2,
                           dateshift, New_Date_1, New_Date_2)
